üéØ CORE STANDARDS
--------------------------------------------------------------------------------
  #pragma once                         - Always use (never #ifndef guards)
  vde_real                             - For all floating-point (never double/float)
  vde_vec3, vde_quat, vde_mat3        - Use math library types
  if (!ptr) return;                    - Always check nulls
  //----------------------             - Use section dividers
  #include "core/math/math_base.h"    - Include in all headers (note: core/math/)

üì¶ KEY TYPES
--------------------------------------------------------------------------------
  vde_real          Configurable precision (double by default)
  vde_vec3          3D vector {x, y, z}
  vde_mat3          3x3 matrix (column-major)
  vde_quat          Quaternion {w, x, y, z}
  vde_frame         Rigid transform {quat q, vec3 p}

üî¢ CONSTANTS

  VDE_RAD2DEG       Radians to degrees multiplier
  VDE_EPS           1e-9 (epsilon for comparisons)
  VDE_SQRT_EPS      1e-6 (sqrt epsilon)

  // Internal State
  //-------------------------

      vde_real data;
  };
  
  //-------------------------
  // Lifecycle
  //-------------------------
  
  Module* module_create(void) {
      Module* m = malloc(sizeof(Module));
      if (!m) return NULL;
      return m;
  }
  
  void module_destroy(Module* m) {
      if (!m) return;
      free(m);
  }

üîç COMMON FUNCTIONS
--------------------------------------------------------------------------------
  Math Helpers:
    vde_abs(x)                    Absolute value
    vde_clamp(x, lo, hi)          Clamp to range
    vde_isfinite(x)               Check for NaN/Inf
    vde_approx(a, b, tol)         Approximate equality
  
  Vector:
    vde_vec3_make(x, y, z)        Create vector
    vde_vec3_add(out, a, b)       Add vectors
    vde_vec3_dot(a, b)            Dot product
    vde_vec3_cross(out, a, b)     Cross product
    vde_vec3_normalize(out, a)    Normalize vector
  
  Quaternion:
    vde_quat_identity()           Identity quaternion
    vde_quat_from_axis_angle()    Create from axis-angle
    vde_quat_rotate(q, v)         Rotate vector
    vde_quat_slerp(q0, q1, t)     Spherical interpolation
  
  Frame:
    vde_frame_identity()          Identity transform
    vde_frame_point_body_to_world()   Transform point
    vde_frame_compose(a, b)       Compose transforms
